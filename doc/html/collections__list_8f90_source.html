674  subroutine list_sort_core(x, fcn)
675  ! Arguments
676  type(container), intent(inout), dimension(:) :: x
677  procedure(compare_items), pointer, intent(in) :: fcn
678 
679  ! Local Variables
680  real(real64) :: r
681  type(container) :: t, tt
682  integer(int32) :: i, ij, j, k, l, m, n, il(21), iu(21)
683 
684  ! Initialization
685  n = size(x)
686 
687  ! Quick Return
688  if (n <= 1) return
689 
690  ! Process
691  m = 1
692  i = 1
693  j = n
694  r = 0.375d0
695  20 if (i == j) go to 60
696  if (r <= 0.5898437d0) then
697  r = r + 3.90625d-2
698  else
699  r = r - 0.21875d0
700  end if
701 
702  30 k = i
703  ! Select a central element of the array, and save it in location T
704  ij = i + int((j - i) * r)
705  t = x(ij)
706 
707  ! Compare: If the first element of the array is greater than T,
708  ! interchange with T
709  if (fcn(x(i)%item, t%item) == 1) then
710  x(ij) = x(i)
711  x(i) = t
712  t = x(ij)
713  end if
714  l = j
715 
716  ! If the last element of the array is less than T, interchange
717  ! with T
718  if (fcn(x(j)%item, t%item) == -1) then
719  x(ij) = x(j)
720  x(j) = t
721  t = x(ij)
722 
723  ! If the first element of the array is greater than T,
724  ! interchange with T
725  if (fcn(x(i)%item, t%item) == 1) then
726  x(ij) = x(i)
727  x(i) = t
728  t = x(ij)
729  end if
730  end if
731 
732  ! Find an element in the second half of the array which is smaller
733  ! than T
734  40 l = l - 1
735  if (fcn(x(l)%item, t%item) == 1) go to 40
736 
737  ! Find an element in the first half of the array which is greater
738  ! than T
739  50 k = k + 1
740  if (fcn(x(k)%item, t%item) == -1) go to 50
741 
742  ! Interchange these elements
743  if (k <= l) then
744  tt = x(l)
745  x(l) = x(k)
746  x(k) = tt
747  go to 40
748  end if
749 
750  ! Save upper and lower subscripts of the array yet to be sorted
751  if (l - i > j - k) then
752  il(m) = i
753  iu(m) = l
754  i = k
755  m = m + 1
756  else
757  il(m) = k
758  iu(m) = j
759  j = l
760  m = m + 1
761  end if
762  go to 70
763 
764  ! Begin again on another portion of the unsorted array
765  60 m = m - 1
766  if (m == 0) go to 100
767  i = il(m)
768  j = iu(m)
769 
770  70 if (j - i >= 1) go to 30
771  if (i == 1) go to 20
772  i = i - 1
773 
774  80 i = i + 1
775  if (i == j) go to 60
776  t = x(i + 1)
777  if (fcn(x(i)%item, t%item) == -1) go to 80
778  k = i
779 
780  90 x(k + 1) = x(k)
781  k = k - 1
782  if (fcn(t%item, x(k)%item) == -1) go to 90
783  x(k + 1) = t
784  go to 80
785 
786  ! End
787  100 continue
788  return
789  end subroutine
